import nltk
from nltk.corpus import stopwords
import string

nltk.download('stopwords')

def clean_text(text):
    text = text.lower()
    text = "".join([char for char in text if char not in string.punctuation])
    
    words = text.split()
    words = [w for w in words if w not in stopwords.words('english')]
    
    return " ".join(words)
